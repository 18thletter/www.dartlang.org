An example of Dart code

class Point  {
  var x, y;
  Point(this.x, this.y);
  scale(factor) => new Point(x*factor, y*factor);
  distance() => Math.sqrt(x*x + y*y);
}

main() {
  var a = new Point(2,3).scale(4);
  print("distance from (0,0): ${a.distance()}");
}
