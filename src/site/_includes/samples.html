int fib(int n, [Map known = const {}]) {
  if (known.containsKey(n))
    return known[n];
  int f = n <= 1 ? n : fib(n - 1, known) + fib(n - 2, known);
  known[n] = f;
  return f;
}

main() => print('${fib(20)}');
