Getters and setters

class Greeter { String _prefix = "Hello, "; String get prefix() => _prefix; //getter for prefix void set prefix(String value) { //setter for prefix if (value == null) value = ""; if (value.length > 20) throw 'Prefix too long!'; _prefix = value; } Greeter() {} greet(name) => print('$prefix $name'); } main() { var greeter = new Greeter(); greeter.prefix = 'Howdy,'; //set prefix greeter.greet('setter!'); }
